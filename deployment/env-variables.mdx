# Environment variables

Các biến môi trường tối thiểu:

\`\`\`bash

# App

NODE_ENV=production

APP_URL=https://homestay.example.com

# Database

DATABASE_URL=postgres://user:pass@host:5432/homestay

# Auth

JWT_SECRET=changeme

JWT_EXPIRES_IN=7d

# Storage

STORAGE_PROVIDER=cloudflare

CLOUDFLARE_R2_ACCESS_KEY=

CLOUDFLARE_R2_SECRET_KEY=

CLOUDFLARE_R2_BUCKET=

# Payment

MOMO_PARTNER_CODE=

MOMO_ACCESS_KEY=

MOMO_SECRET_KEY=

VNPAY_TMN_CODE=

VNPAY_HASH_SECRET=

# Email
